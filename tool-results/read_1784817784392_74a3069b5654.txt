   320‚Üí   320‚Üíasync def verify_payment(tx_hash: str) -> dict:
   321‚Üí   321‚Üí    tx_hash_clean = tx_hash.strip()
   322‚Üí   322‚Üí    if tx_hash_clean.startswith("0x"):
   323‚Üí   323‚Üí        tx_hash_clean = tx_hash_clean[2:]
   324‚Üí   324‚Üí    if len(tx_hash_clean) != 64:
   325‚Üí   325‚Üí        return {"valid": False, "error": "Invalid transaction hash format.", "details": {}}
   326‚Üí   326‚Üí    tx_hash_full = f"0x{tx_hash_clean}"
   327‚Üí   327‚Üí    if tx_hash_full in used_transactions:
   328‚Üí   328‚Üí        return {"valid": False, "error": "Transaction already used.", "details": {}}
   329‚Üí   329‚Üí    try:
   330‚Üí   330‚Üí        tx_receipt = await get_tx_receipt(tx_hash_full)
   331‚Üí   331‚Üí        if not tx_receipt:
   332‚Üí   332‚Üí            return {"valid": False, "error": "Transaction not found or not confirmed.", "details": {}}
   333‚Üí   333‚Üí    except Exception as e:
   334‚Üí   334‚Üí        return {"valid": False, "error": f"Transaction not found: {str(e)}", "details": {}}
   335‚Üí   335‚Üí    if int(tx_receipt.get("status", "0x0"), 16) != 1:
   336‚Üí   336‚Üí        return {"valid": False, "error": "Transaction failed (reverted).", "details": {}}
   337‚Üí   337‚Üí    wallet_lower = WALLET_ADDRESS.lower()
   338‚Üí   338‚Üí    expected_amount_wei = int(float(PRICE_USDC) * 10**6)
   339‚Üí   339‚Üí    found = False
   340‚Üí   340‚Üí    transfer_details = {}
   341‚Üí   341‚Üí    for log in tx_receipt.get("logs", []):
   342‚Üí   342‚Üí        if log["address"].lower() != USDC_CONTRACT.lower():
   343‚Üí   343‚Üí            continue
   344‚Üí   344‚Üí        topics = log.get("topics", [])
   345‚Üí   345‚Üí        if len(topics) < 3:
   346‚Üí   346‚Üí            continue
   347‚Üí   347‚Üí        if topics[0] != TRANSFER_TOPIC:
   348‚Üí   348‚Üí            continue
   349‚Üí   349‚Üí        from_addr = "0x" + topics[1][-40:]
   350‚Üí   350‚Üí        to_addr = "0x" + topics[2][-40:]
   351‚Üí   351‚Üí        amount = int(log["data"], 16)
   352‚Üí   352‚Üí        if to_addr.lower() != wallet_lower:
   353‚Üí   353‚Üí            continue
   354‚Üí   354‚Üí        if amount == expected_amount_wei:
   355‚Üí   355‚Üí            found = True
   356‚Üí   356‚Üí            transfer_details = {
   357‚Üí   357‚Üí                "from": from_addr, "to": to_addr,
   358‚Üí   358‚Üí                "amount_raw": amount, "amount_usdc": amount / 10**6,
   359‚Üí   359‚Üí                "block": int(log["blockNumber"], 16), "tx_hash": tx_hash_full
   360‚Üí   360‚Üí            }
   361‚Üí   361‚Üí            break
   362‚Üí   362‚Üí    if not found:
   363‚Üí   363‚Üí        return {"valid": False, "error": f"No valid USDC transfer of exactly {PRICE_USDC} USDC found.", "details": {}}
   364‚Üí   364‚Üí    return {"valid": True, "error": None, "details": transfer_details}
