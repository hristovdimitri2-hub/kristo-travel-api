   258‚Üí   258‚Üíasync def verify_payment(tx_hash: str) -> dict:
   259‚Üí   259‚Üí    tx_hash_clean = tx_hash.strip()
   260‚Üí   260‚Üí    if tx_hash_clean.startswith("0x"):

   262‚Üí   262‚Üí    if len(tx_hash_clean) != 64:
   263‚Üí   263‚Üí        return {"valid": False, "error": "Invalid transaction hash format.", "details": {}}
   264‚Üí   264‚Üí    tx_hash_full = f"0x{tx_hash_clean}"
   265‚Üí   265‚Üí    if tx_hash_full in used_transactions:
   266‚Üí   266‚Üí        return {"valid": False, "error": "Transaction already used.", "details": {}}
   267‚Üí   267‚Üí    try:

   270‚Üí   270‚Üí    except Exception as e:
   271‚Üí   271‚Üí        return {"valid": False, "error": f"Transaction not found or not confirmed: {str(e)}", "details": {}}
   272‚Üí   272‚Üí    if tx_receipt["status"] != 1:
   273‚Üí   273‚Üí        return {"valid": False, "error": "Transaction failed (reverted).", "details": {}}
   274‚Üí   274‚Üí    wallet_lower = WALLET_ADDRESS.lower()
   275‚Üí   275‚Üí    expected_amount_wei = int(float(PRICE_USDC) * 10**6)
   276‚Üí   276‚Üí    found_valid_transfer = False
   277‚Üí   277‚Üí    transfer_details = {}
   278‚Üí   278‚Üí    for log in tx_receipt.get("logs", []):
   279‚Üí   279‚Üí        if log["address"].lower() != USDC_CONTRACT.lower():
   280‚Üí   280‚Üí            continue
   281‚Üí   281‚Üí        if len(log.get("topics", [])) < 3:
   282‚Üí   282‚Üí            continue
   283‚Üí   283‚Üí        if log["topics"][0].hex() != TRANSFER_TOPIC:
   284‚Üí   284‚Üí            continue
   285‚Üí   285‚Üí        from_address = "0x" + log["topics"][1].hex()[-40:]
   286‚Üí   286‚Üí        to_address = "0x" + log["topics"][2].hex()[-40:]
   287‚Üí   287‚Üí        amount = int(log["data"].hex(), 16)
   288‚Üí   288‚Üí        if to_address.lower() != wallet_lower:
   289‚Üí   289‚Üí            continue

   291‚Üí   291‚Üí            found_valid_transfer = True
   292‚Üí   292‚Üí            transfer_details = {
   293‚Üí   293‚Üí                "from": from_address, "to": to_address,
   294‚Üí   294‚Üí                "amount_raw": amount, "amount_usdc": amount / 10**6,
   295‚Üí   295‚Üí                "block": tx_receipt["blockNumber"], "tx_hash": tx_hash_full
   296‚Üí   296‚Üí            }
   297‚Üí   297‚Üí            break
   298‚Üí   298‚Üí    if not found_valid_transfer:
   299‚Üí   299‚Üí        return {"valid": False, "error": f"No valid USDC transfer of exactly {PRICE_USDC} USDC found.", "details": {}}
   300‚Üí   300‚Üí    return {"valid": True, "error": None, "details": transfer_details}
